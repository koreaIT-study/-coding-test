``` java
package cote.baekjun;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;

public class Editer {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder line = new StringBuilder(br.readLine());
        int N = Integer.parseInt(br.readLine());

        int cursor = line.length(); // 초기 커서의 위치

        while (N-- > 0) {
            StringBuilder commandLine = new StringBuilder(br.readLine());
            char c = commandLine.charAt(0);

            switch (c) {
            case 'L':
                if (cursor != 0)
                    cursor--;
                break;
            case 'D':
                if (cursor != line.length())
                    cursor++;
                break;
            case 'B':
                if (cursor != 0) {
                    line.delete(cursor - 1, cursor);
                    cursor--;
                }
                break;
            case 'P':
                line.insert(cursor, commandLine.substring(2));
                cursor++; // 커서 왼쪽이므로 커서도 다시 ++해줘야함
                break;
            default:
                bw.write("잘못입력");
                bw.flush();
                bw.close();
                br.close();
                return;
            }
        }

        bw.write(line.toString());
        bw.flush();
        bw.close();
        br.close();
    }
}
```  

``` java
package cote.baekjun;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.IOException;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.LinkedList;

public class Editer {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

        LinkedList<Character> list = new LinkedList<>();
        StringBuilder sb = new StringBuilder();
        
        for (char c : br.readLine().toCharArray()) {
            list.add(c);
        }

        int N = Integer.parseInt(br.readLine());
        int cursor = list.size(); // 초기 커서의 위치

        while (N-- > 0) {
            StringBuilder commandLine = new StringBuilder(br.readLine());
            char c = commandLine.charAt(0);

            switch (c) {
            case 'L':
                if (cursor != 0)
                    cursor--;
                break;
            case 'D':
                if (cursor != list.size())
                    cursor++;
                break;
            case 'B':
                if (cursor != 0) {
                    list.remove(cursor-1);
                    cursor--;
                }
                break;
            case 'P':
                list.add(cursor, commandLine.charAt(2));
                cursor++; // 커서 왼쪽이므로 커서도 다시 ++해줘야함
                break;
            default:
                bw.write("잘못입력");
                bw.flush();
                bw.close();
                br.close();
                return;
            }
        }

        for(Character c : list) {
            sb.append(c);
        }
        bw.write(sb.toString());
        bw.flush();
        bw.close();
        br.close();
    }
}
```
