import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.EmptyStackException;

public class Main {

	private static java.util.Stack<Integer> stack = new java.util.Stack<>();
	
	private static void solution(String line) {
		
		String[] command = line.split(" ");
		int stackData = 0;
		if(command.length != 1) {
			stackData = Integer.parseInt(command[1]);
		}
		
		print(command[0], stackData);
	}

	private static void print(String method, int stackData) {
		switch (method) {
		case "push":
			stack.push(stackData);
			break;
		case "pop":
			try {
				System.out.println(stack.pop());
			}catch(EmptyStackException e) {
				System.out.println(-1);
			}
			break;
		case "empty":
			if(stack.isEmpty())
				System.out.println(1);
			else 
				System.out.println(0);
			break;
		case "top":
			try {
				System.out.println(stack.peek());
			}catch(EmptyStackException e) {
				System.out.println(-1);
			}
			break;
		case "size":
			System.out.println(stack.size());
			break;
		default:
			System.out.println("잘못 들어옴");
			break;
		}
	}
	
	public static void main(String[] args) {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		
		try {
			int N = Integer.parseInt(br.readLine());
			for (int i = 0; i < N; i++) {
				solution(br.readLine());
			}
		} catch (IOException e) {
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
	}

}
